<?xml version="1.0"?>
<project default="test">

<target name="test">
 <r-set property="rhome" code="R.home()" />
 <r-set property="rversion" code="version$version.string" />

 <r-run code="rjavaversion = packageDescription( 'rJava', fields='Version' )" />
 <r-set property="rjavahome" code="system.file( package = 'rJava' )" />
 <r-set property="rjavaversion" code="rjavaversion" />
 
 <echo>
    R home        : ${rhome}
    R version     : ${rversion}
    rJava home    : ${rjavahome}
    rJava version : ${rjavaversion}
 </echo>

</target>

</project>
